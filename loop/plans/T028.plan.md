// apps/api/test/factories/index.ts — ADD (do not alter existing exports)
import type { Pet, Species } from "@prisma/client";   // add to the existing @prisma/client type import

export async function createPet(
  prisma: PrismaClient,
  householdId: string,
  overrides?: { species?: Species; name?: string },
): Promise<Pet> {
  return prisma.pet.create({
    data: {
      householdId,
      species: overrides?.species ?? "DOG",
      name: overrides?.name ?? "Fido",
    },
  });
}

// Wrap createAuthedContext + userIds bookkeeping (the duplicated owner()/member()):
export async function createOwnerContext(
  app: INestApplication, prisma: PrismaClient, jwt: JwtService, userIds: string[],
): Promise<AuthedContext> {
  const ctx = await createAuthedContext(app, prisma, jwt, { role: "OWNER" });
  userIds.push(ctx.user.id);
  return ctx;
}
export async function createMemberContext(
  app: INestApplication, prisma: PrismaClient, jwt: JwtService, userIds: string[],
): Promise<AuthedContext> {
  const ctx = await createAuthedContext(app, prisma, jwt, { role: "MEMBER" });
  userIds.push(ctx.user.id, ctx.household.ownerId);
  return ctx;
}

## Risks & the design decisions the planner made (for CHECKER scrutiny)
1. **`createPet` is prisma-level `(prisma, householdId, overrides?)` (matches the card's stated signature), adopted at the ONE raw `prisma.pet.create` (households ~line 202).** photos' local `createPet(ctx, name)` is left HTTP-based (it deliberately drives the presign prerequisite through the API) — I did NOT convert it to prisma, because changing a pet's creation mechanism is a behavioral setup change, not a mechanical refactor. Consequence: `createPet` has one adoption site, below the "≥2 suites" guidance. Chosen over forcing 2 adoptions to keep the refactor strictly mechanical and assertion-preserving. Checker: confirm the households seed line is the only createPet swap and no assertion moved.
2. **The real ≥3-suite consolidation is `createOwnerContext`/`createMemberContext`** (the verbatim-duplicated `owner()`/`member()` in pets/photos/households — the "household factories" the card asks for). Delegation keeps call sites byte-identical. Alternative (inlining `createAuthedContext` at each call site) rejected: would churn assertions and lose the userIds bookkeeping. Checker: diff must show identical `it(...)` titles/counts.
3. **`createInvite` factory NOT added.** Invites appear only in `households.e2e` and are minted via the owner-only endpoint that is itself under test (plus one `.update`/`.findUnique`); there is no raw `prisma.householdInvite.create` to consolidate. Adding a seed factory would be unused — over-engineering (CLAUDE §2 rule 2). Documented so the checker doesn't read the omission as an oversight.
4. **No `apps/api/package.json` change.** The T020 `collectCoverageFrom: ["src/**/*.service.ts"]` glob already instruments all five P2 services; the card's "confirm inside the glob" is a verification, not a config edit. `images.processor.ts` stays outside by design. Risk: if a future service sits outside `.service.ts` it would be silently uncovered — not the case for P2.
5. **Straggler-3 (offline snapshot) skipped, not added.** Lowest-risk path (avoids shared-store fragility in the snapshot suite); offline is already assertion-covered. Trade-off: the T025 note is closed by rationale rather than a new snapshot. If the checker deems a snapshot mandatory, it is a one-`it` follow-up — flagged here so the decision is visible, not silent.
6. **Straggler-2 (worker-exit warning) may end unresolved-but-journaled.** If the leak is in `src` lifecycle, this test-only task cannot fix it; journaling a known pre-existing warning is the correct outcome, consistent with the card ("fix only if a one-liner, else journal as known").
7. **Conditional gap-fillers enumerated but expected unused.** The five P2 services are heavily exercised by their e2e suites, so `test:cov` likely already passes; the spec files (6–8) are authorized only to make the AC deterministically reachable, and only the exact sub-threshold service may be added.
