export const SIZE_CLASSES = ["TOY", "SMALL", "MEDIUM", "LARGE", "GIANT"] as const;
export type SizeClass = (typeof SIZE_CLASSES)[number];

// Structured-sign vocabulary the app collects; T032's richer intake maps INTO this (contract R2).
export const RED_FLAG_SIGNS = [
  "toxin_ingestion","chocolate_ingestion","xylitol_ingestion","rodenticide_exposure",
  "retching_unproductive","distended_abdomen","straining_to_urinate","blood_in_urine",
  "seizure_prolonged_or_repeated","collapse_unresponsive","abnormal_gum_color",
  "breathing_difficulty","uncontrolled_bleeding","heatstroke","envenomation","major_trauma",
  "ocular_emergency","unable_to_stand","linear_foreign_body","open_fracture_or_deep_wound",
  "birthing_distress","neonatal_collapse",
] as const;
export type RedFlagSign = (typeof RED_FLAG_SIGNS)[number];

// Minimal input shape (T032 will map its intake into this — R2). Every field except species is optional.
export interface RedFlagIntake {
  species: Species;                                  // always present
  sex?: Sex;                                         // MALE | FEMALE | UNKNOWN
  ageMonths?: number;
  weightKg?: number;
  sizeClass?: SizeClass;
  signs?: Partial<Record<RedFlagSign, boolean>>;     // structured checkboxes
  freeText?: string;                                 // owner's typed description
}

// ---- Predicate / matcher DSL (data-driven for auditability — R3) ----
export type IntakePredicate =
  | { field: "species"; op: "eq"; value: Species }
  | { field: "species"; op: "in"; value: readonly Species[] }
  | { field: "sex"; op: "in"; value: readonly Sex[] }       // undefined sex ⇒ MATCH (fail-upward, R6)
  | { field: "ageMonths"; op: "lte" | "gte"; value: number }// undefined ⇒ no match
  | { field: "weightKg"; op: "lte" | "gte"; value: number } // undefined ⇒ no match
  | { field: "sign"; sign: RedFlagSign };                   // matches iff signs[sign] === true
export interface KeywordLeaf { kw: readonly string[] }      // matches iff normalizedText contains ANY phrase
export type Matcher = IntakePredicate | KeywordLeaf | { allOf: readonly Matcher[] } | { anyOf: readonly Matcher[] };

export interface RedFlagRule {
  id: string;                       // kebab-case, unique
  species: Species | "ANY";
  tierFloor: Urgency;               // v1: all EMERGENCY_NOW (field future-proofs VET_24H — R7)
  emergencyPayloadKey: string;      // stable key T036 consumes for interstitial content
  label: string;                    // neutral internal descriptor for logs/evals (§7-clean)
  sourceRef: string;                // "SPEC §6.2: <verbatim>" or "extra: <justification>" (audit trail)
  match: Matcher;
}

export interface RedFlagMatch {
  ruleId: string;
  species: Species | "ANY";
  tierFloor: Urgency;
  emergencyPayloadKey: string;
  label: string;
}
export interface RedFlagEvaluation {
  matched: RedFlagMatch[];          // all matches, sorted most-urgent first
  highest: RedFlagMatch | null;     // matched[0] ?? null
}

// engine.ts
export function evaluateRedFlags(intake: RedFlagIntake): RedFlagEvaluation;

// normalize.ts
export function normalizeText(raw: string): string;

### Matcher semantics (executor must implement precisely)
- A rule fires iff `evalMatcher(rule.match, intake, normalizeText(intake.freeText ?? ""))` is `true`, AND `rule.species === "ANY" || rule.species === intake.species` (species gate applied first, cheap short-circuit).
- `KeywordLeaf.kw`: matches iff the normalized free text contains AT LEAST ONE phrase as a substring (phrases are stored pre-normalized). Empty free text ⇒ no keyword leaf matches.
- `{ field: "sign", sign }`: matches iff `intake.signs?.[sign] === true`. Absent ⇒ false (the keyword channel supplies the fail-upward path from text).
- `{ field: "sex", op: "in", value }`: matches iff `intake.sex === undefined || value.includes(intake.sex)` (fail-upward on unknown sex — R6, safety-load-bearing).
- `{ field: "species", ... }`: species is always present.
- numeric `ageMonths/weightKg lte|gte`: matches only when the field is defined AND satisfies the comparison; undefined ⇒ false.
- `allOf`: every child true. `anyOf`: at least one child true. Combinators nest arbitrarily.
- Engine evaluates ALL rules (never short-circuits the table), collects matches, sorts by `URGENCY_SEVERITY[tierFloor]` ascending (0 = most urgent) with a stable tie-break by table order (preserve declaration order for equal severity). `highest = matched[0] ?? null`.

## Tests to write (map to acceptance criteria) — counts to ≥60
**AC "≥60 table-driven unit tests (positives, near-miss negatives, normalization)"** — the per-rule matrix alone is 66 cases.

- **AC per-rule (66 cases) → `engine.spec.ts` › "each rule: structured positive, keyword positive, near-miss negative"** using `it.each(RULE_CASES)` where `RULE_CASES` is a table with one entry per rule id supplying `{structuredIntake, keywordIntake, nearMissIntake}`:
  - structured positive: minimal intake (correct species + the rule's structured `signs` true, and correct `sex` where gated) ⇒ `highest.ruleId === id`, `tierFloor === "EMERGENCY_NOW"`.
  - keyword positive: minimal intake (correct species/sex + `freeText` containing one phrase from the rule's kw group) ⇒ rule present in `matched` (assert `matched.some(m => m.ruleId === id)`; for multi-channel rules like GDV, freeText must contain BOTH a RETCH and a BLOAT phrase).
  - near-miss negative: an intake that is close but must NOT fire this rule (e.g. correct species, a *sibling* sign/phrase that is not sufficient, or wrong species) ⇒ `matched.every(m => m.ruleId !== id)`. Near-misses are chosen to contain NO trigger phrase (honest, per D2) — e.g. `distended-abdomen` near-miss freeText "eating normally, belly looks fine".
  - 22 rules × 3 = **66 tests**.
- **AC named case "male cat straining" → `engine.spec.ts` › "male cat straining to urinate"** (4 asserts):
  - MALE cat + `sign straining_to_urinate` ⇒ `urinary-blockage-cat` fires (EMERGENCY_NOW).
  - UNKNOWN-sex cat + freeText "straining to pee" ⇒ fires (fail-upward, R6).
  - MALE cat + freeText "cant pee, in and out of litter box" ⇒ fires.
  - FEMALE cat + freeText "straining to pee" (no blood) ⇒ `urinary-blockage-cat` does NOT fire (near-miss; documents the sex gate) — and neither does `urinary-obstruction-signs-cat` (no blood). Assert `highest === null`.
- **AC named case "retching + bloated large dog" → `engine.spec.ts` › "retching + distended abdomen dog (GDV)"** (5 asserts):
  - LARGE dog + signs retching_unproductive + distended_abdomen ⇒ `gdv-suspected` fires.
  - dog + freeText "dry heaving and his belly is swollen and hard" ⇒ `gdv-suspected` fires.
  - SMALL dog + both signs ⇒ still fires (size-agnostic — D-size); assert fires.
  - retching-only (freeText "dry heaving") ⇒ `gdv-suspected` does NOT fire AND `distended-abdomen` does NOT fire ⇒ `highest === null` (honest near-miss).
  - bloat-only (freeText "his belly is swollen") ⇒ `gdv-suspected` does NOT fire BUT `distended-abdomen` DOES fire (documents the intentional rule-18 split, R8).

- **species gating → `engine.spec.ts` › "species gating"** (≥3): a DOG-only rule (`gdv-suspected`) never fires for a CAT with the same signs; a CAT-only rule (`urinary-blockage-cat`) never fires for a DOG; an ANY rule fires for both species.

## Risks & design decisions the planner made (for CHECKER scrutiny)
- **R1 — `packages/ai` gains a `@pawcareright/types` workspace dependency (+ `pnpm-lock.yaml` regen).** Required to type `tierFloor: Urgency` and sort by `URGENCY_SEVERITY` from the canonical T030 source (CLAUDE "never hand-write a duplicate interface"). Internal workspace dep, not a third-party package. If the checker flags an unexpected `package.json`/lockfile change: it is deliberate and listed here; the lock diff must be workspace-only (no new external packages).
- **R2 — `RedFlagIntake` is a minimal, NON-zod input shape defined HERE (T031 depends only on T005).** T032's richer intake will MAP into it (`species`, `sex`, `ageMonths`, `weightKg`, `sizeClass`, `signs: Partial<Record<RedFlagSign,boolean>>`, `freeText`). The `RedFlagSign` union IS the mapping contract T032/T034 must satisfy. Trade-off: two shapes to keep in sync — mitigated by making this the single exported vocabulary T032 imports.
- **R3 — Data-driven predicate DSL (`{field,op,value}` + allOf/anyOf + kw leaf), NOT arbitrary per-rule functions.** More verbose but auditable/serializable — the table is the C1 vet-review artifact; a reviewer reads data, not closures. This is the primary architecture decision.
- **R4 — Rule fires on structured-sign OR free-text keyword (composed with allOf/anyOf), gated inside the same matcher.** Lets one rule cover both the checkbox path and the typed path (the AC's "structured positive" + "keyword positive"), while keeping species/sex gates binding on BOTH channels (so the FEMALE-cat-straining near-miss stays a negative).
- **R5 — Normalization is hand-rolled (~40 lines), NO stemming/NLP dependency (CLAUDE §2 rule 7).** NFKD + diacritic strip + lowercase + punctuation→space + collapse. Multilingual-SAFE for v1 English; non-Latin scripts pass through (no crash) but v1 keyword sets are English (documented).
- **R6 — Fail-upward on unknown/absent `sex` for the male-cat gate.** `sex in [MALE,UNKNOWN]` matches when `intake.sex` is `undefined` — a cat straining with unrecorded sex still triggers the urethral-obstruction emergency (can't rule out male). Explicit `FEMALE` is excluded (near-miss negative). This is the single most safety-load-bearing predicate — scrutinize.
- **R7 — All 22 v1 rules are `EMERGENCY_NOW`; `tierFloor` field future-proofs `VET_24H`.** Keeps the table's contract crisp ("this table = near-certain emergencies"); sub-emergency tiering (incl. female-straining, mild cases) is the AI layer's job (T033/T034). Deliberately did NOT add a `VET_24H` female-straining rule here — mixing floors would dilute the deterministic-emergency contract.
- **R8 — Intentional overlap: `gdv-suspected` (retch+bloat, dog) vs `distended-abdomen` (bloat-alone, any).** Distinct payload keys ⇒ distinct T036 interstitials; both EMERGENCY. Bloat-alone deliberately still fires (rule 18), so the GDV near-miss test is "retching-only" (a true negative), not "bloat-only".
- **D2 — Over-trigger bias, negation-free v1.** Keyword phrases are distress-specific (never bare organ words), so "breathing normally" does not match; but no negation handling means "not vomiting" could over-trigger — accepted (false-positive emergency is safe; false-negative is not — SPEC §5 rule 2). Documented; a couple of honest near-miss negatives assert the boundary.
- **D-perf — AC "<5ms per evaluation" verified as steady-state average over N=1000 on a worst-case input**, not single-call p100 (JIT/GC-flaky). Real cost is sub-ms (~50× headroom), so the average bound is both faithful to the AC and CI-stable. If the checker prefers p95, it is trivially also < 5ms given the headroom.
- **D-catbias — the global "cats one tier more urgent" bias (SPEC §5 rule 2) is NOT in this table.** Rules are explicitly per-species where the medicine differs; the blanket cat up-tiering is applied by the AI post-rules tiering (T033/T034), never here. A dedicated test asserts the engine does not up-tier cats. Flagged for the checker as the boundary decision the card asked to pin down.
